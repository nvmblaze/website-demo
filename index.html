CONGESTION CONTROL USING NS
write a program to Simulate Congestion Control Algorithms using NS?


# Create Simulator
set ns [new Simulator]

# Trace files
set tracefile [open congestion.tr w]
$ns trace-all $tracefile

set namfile [open congestion.nam w]
$ns namtrace-all $namfile

# Create Nodes
set n0 [$ns node]   ;# Sender 1
set n1 [$ns node]   ;# Router
set n2 [$ns node]   ;# Receiver
set n3 [$ns node]   ;# Sender 2 (for congestion)

# Create Links (bottleneck at n1)
$ns duplex-link $n0 $n1 2Mb 10ms DropTail
$ns duplex-link $n3 $n1 2Mb 10ms DropTail
$ns duplex-link $n1 $n2 1Mb 20ms DropTail   ;# Bottleneck link

# Queue size (to force congestion)
$ns queue-limit $n1 $n2 10

# -------------------------------
# TCP Agent (CHANGE HERE)
# -------------------------------
# Options:
# Agent/TCP        -> Tahoe
# Agent/TCP/Reno   -> Reno
# Agent/TCP/Newreno -> New Reno

set tcp1 [new Agent/TCP/Reno]
$tcp1 set class_ 1
$ns attach-agent $n0 $tcp1

set sink1 [new Agent/TCPSink]
$ns attach-agent $n2 $sink1

$ns connect $tcp1 $sink1

# Second TCP flow (to create congestion)
set tcp2 [new Agent/TCP/Reno]
$ns attach-agent $n3 $tcp2

set sink2 [new Agent/TCPSink]
$ns attach-agent $n2 $sink2

$ns connect $tcp2 $sink2

# Applications (FTP traffic)
set ftp1 [new Application/FTP]
$ftp1 attach-agent $tcp1

set ftp2 [new Application/FTP]
$ftp2 attach-agent $tcp2

# Start/Stop Times
$ns at 0.5 "$ftp1 start"
$ns at 1.0 "$ftp2 start"

$ns at 8.0 "$ftp1 stop"
$ns at 8.5 "$ftp2 stop"

# Finish Procedure
proc finish { } {
    global ns tracefile namfile
    $ns flush-trace
    close $tracefile
    close $namfile
    exec nam congestion.nam &
    exit 0
}

$ns at 9.0 "finish"

# Run Simulation
$ns run
